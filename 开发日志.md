### 开启开发环境
# 在树莓派VNC桌面中：
# 1. 打开终端（黑色图标）
# 2. 激活虚拟环境
source ~/myenv/bin/activate

# 3. 进入项目目录
cd ~/smart_rescue

# 4. 打开代码编辑器
geany src/vision_code.py

————————————————————————————————————————————————————
### 编写与测试代码
# 在编辑器中写代码，然后在终端测试
# 运行当前代码
python src/vision_code.py

# 如果摄像头被占用，先释放
sudo pkill -f python

————————————————————————————————————————————————————————————
### 保存与备份
# 每日结束前备份代码
cp src/vision_code.py src/backups/vision_$(date +%Y%m%d).py
